def matrix_to_weight_network(matrix,ARP_list,node_weight): #matrix weight network 
    G = nx.Graph()
    G.add_nodes_from([("ARP{}".format(i+1), {'weight': node_weight[i]}) for i in range(len(ARP_list))])
    for i in range(len(ARP_list)-1):
        for j in range(i + 1, len(ARP_list)):  
            if (matrix[i, j] != 0):
                G.add_edge(ARP_list[i], ARP_list[j],weight=matrix[i,j])  # weight    
    return G
